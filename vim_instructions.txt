" Set tab size to 4 spaces, but use actual tab characters instead of spaces
set tabstop=4
set shiftwidth=4
set softtabstop=4
set noexpandtab

" Render whitespace characters (tabs, spaces, end of line, etc.)
set list
set listchars=tab:▸\ ,trail:·,extends:>,precedes:<,nbsp:␣

" Trim trailing whitespace automatically on save
autocmd BufWritePre * :%s/\s\+$//e

" Disable automatic addition of a newline at the end of the file
" Commenting out or removing the previous autocmd for final newline

" Enable syntax highlighting
syntax on

" Enable line numbers (static, no relative line numbers)
set number
set norelativenumber

" Highlight search results
set hlsearch

" Enable incremental search
set incsearch

" Set color scheme (adjust as needed)
colorscheme desert

" Enable file type detection, plugin, and indent support
filetype plugin indent on

" Display the current command in the status line
set showcmd

" Always show the status line
set laststatus=2

" Enable auto-indentation
set autoindent
set smartindent

" Enable mouse support (useful for scrolling and text selection)
set mouse=a

" Use system clipboard for copy-paste operations
set clipboard=unnamedplus

" Display a sign column next to the line numbers
set signcolumn=yes

" Enable backspace to work on every character in insert mode
set backspace=indent,eol,start

" Display matching parentheses, brackets, and braces
set showmatch

" Disable line wrapping to prevent long lines from breaking
set nowrap

" Use a fixed-width font to align code properly
set guifont=Monospace:h12

" Automatically reload files that have been changed outside of Vim
set autoread
autocmd FocusGained,BufEnter * checktime

" Set the maximum text width for line wrapping (80 characters for 42 norm)
set textwidth=80
set colorcolumn=+1

" Show the cursor position in the status line
set ruler

" Split windows below and to the right (makes navigation more intuitive)
set splitbelow
set splitright

" Enable persistent undo (so you can undo changes even after closing and reopening Vim)
set undofile
